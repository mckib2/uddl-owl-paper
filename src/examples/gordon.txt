(AirSystem, composes, Identifier,)
(AirSystem, composes AirFrame, airframe)
(AirSystem, composes, NavigationSystem, navSystem)

(AirFrame, composes, Identifier,)
(Airframe, composes, Position, pos)
(Airframe, composes, Velocity, vel)
(Airframe, composes, Acceleration, accel)

(NavigationSystem, composes, Identifier,)
(NavigationSystem, composes, Position, pos)
(NavigationSystem, composes, Mode, mode)

(Observe, composes, Identifier,)
(Observe, composes, CalendarTime, time)
(Observe, composes, Validity, validity)
(Observe, associates[1, 1][1, 1], AirFrame, observed)
(Observe, associates[1, 1][1, 1], NavigationSystem, observer)

SELECT AirFrame.pos, NavigationSystem.mode, Observe.validity
FROM AirSystem
JOIN NavigationSystem ON AirSystem.navSystem
JOIN Observe ON Observe.observer = NavigationSystem
JOIN AirFrame ON Observe.observed AND AirSystem.airFrame;
